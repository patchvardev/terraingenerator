<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<script>

    // --- 1. THE GEOMETRY ENGINE ---

    let scene, camera, renderer, terrain, geometry;

    let seed = Math.random() * 100;


    function init() {

        scene = new THREE.Scene();

        camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);

        renderer = new THREE.WebGLRenderer({ antialias: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        document.body.appendChild(renderer.domElement);


        camera.position.set(0, 40, 60);

        camera.lookAt(0, 0, 0);


        const light = new THREE.DirectionalLight(0x00ffcc, 1);

        light.position.set(10, 50, 10);

        scene.add(light);

        scene.add(new THREE.AmbientLight(0x404040));


        createTerrain();

        animate();

    }


    function createTerrain() {

        if (terrain) scene.remove(terrain);

        

        const res = 60;

        geometry = new THREE.PlaneGeometry(100, 100, res, res);

        const pos = geometry.attributes.position;

        const octaves = parseInt(document.getElementById('octaves').value);

        const amp = parseFloat(document.getElementById('amp').value);


        for (let i = 0; i < pos.count; i++) {

            let x = pos.getX(i) * 0.05;

            let y = pos.getY(i) * 0.05;

            let z = 0;

            

            // Layered Noise (Instructional Logic)

            for(let j = 1; j <= octaves; j++) {

                z += (Math.sin(x * j + seed) * Math.cos(y * j + seed)) * (amp / j);

            }

            pos.setZ(i, z);

        }

        

        geometry.computeVertexNormals();

        const material = new THREE.MeshPhongMaterial({ 

            color: 0x111111, 

            wireframe: true, 

            emissive: 0x00ffcc,

            emissiveIntensity: 0.2 

        });

        terrain = new THREE.Mesh(geometry, material);

        terrain.rotation.x = -Math.PI / 2;

        scene.add(terrain);

    }


    // --- 2. THE SONIFICATION ENGINE ---

    const AudioContext = window.AudioContext || window.webkitAudioContext;

    let audioCtx;


    function playSound() {

        if (!audioCtx) audioCtx = new AudioContext();

        const base = parseFloat(document.getElementById('freq').value);

        const octaves = parseInt(document.getElementById('octaves').value);

        

        // Pentatonic Scale "Language"

        const scale = [1, 1.125, 1.25, 1.5, 1.66]; 

        

        for (let i = 0; i < octaves; i++) {

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            

            // Map terrain features to notes

            const noteIndex = Math.floor(Math.random() * scale.length);

            const freq = base * scale[noteIndex] * (i + 1);

            

            osc.type = i % 2 === 0 ? 'sine' : 'triangle';

            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);

            

            gain.gain.setValueAtTime(0, audioCtx.currentTime);

            gain.gain.linearRampToValueAtTime(0.1, audioCtx.currentTime + 0.5);

            gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 3);


            osc.connect(gain);

            gain.connect(audioCtx.destination);

            

            osc.start();

            osc.stop(audioCtx.currentTime + 3);

        }

    }


    // --- 3. THE COLLECTIVE INTERFACE ---

    document.getElementById('trigger').addEventListener('click', playSound);

    document.getElementById('random').addEventListener('click', () => {

        seed = Math.random() * 100;

        createTerrain();

    });

    document.querySelectorAll('input').forEach(i => i.addEventListener('input', createTerrain));


    function animate() {

        requestAnimationFrame(animate);

        terrain.rotation.z += 0.002;

        renderer.render(scene, camera);

    }


    init();

</script>
